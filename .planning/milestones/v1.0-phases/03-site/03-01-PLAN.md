    **Chart.js script block** (separate from data injection, no define:vars):
    ```astro
    <script>
      import Chart from 'chart.js/auto';

      const raw = JSON.parse(document.getElementById('chart-data')!.textContent!);
      const months = raw.months as Array<{ month: string; pensionEur: number; scamEur: number | null }>;

      const labels = months.map((m: any) => {
        const [year, mon] = m.month.split('-');
        const MONTH_SHORT: Record<string, string> = {
          '01':'jaan','02':'veebr','03':'märts','04':'apr','05':'mai','06':'juuni',
          '07':'juuli','08':'aug','09':'sept','10':'okt','11':'nov','12':'dets'
        };
        return `${MONTH_SHORT[mon] ?? mon} ${year}`;
      });

      const canvas = document.getElementById('chart') as HTMLCanvasElement;
      new Chart(canvas, {
        type: 'line',
        data: {
          labels,
          datasets: [
            {
              label: 'Pensionimaksed',
              data: months.map((m: any) => m.pensionEur === 0 ? null : m.pensionEur),
              borderColor: '#2563eb',
              backgroundColor: 'transparent',
              pointRadius: 4,
              spanGaps: true,
            },
            {
              label: 'Pettusekahju',
              data: months.map((m: any) => m.scamEur),
              borderColor: '#dc2626',
              backgroundColor: 'transparent',
              pointRadius: 4,
              spanGaps: true,
            }
          ]
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          interaction: { mode: 'index', intersect: false },
          plugins: {
            tooltip: {
              callbacks: {
                label: (ctx: any) => {
                  const val = ctx.raw as number | null;
                  if (val === null) return `${ctx.dataset.label}: andmed puuduvad`;
                  return `${ctx.dataset.label}: ${val.toLocaleString('et-EE')} €`;
                }
              }
            }
          },
          scales: {
            y: {
              beginAtZero: true,
              ticks: {
                callback: (v: any) => `${(v / 1_000_000).toFixed(0)} M€`
              }
            }
          }
        }
      });

      // Signal canvas is ready for OG screenshot
      canvas.addEventListener('animationend', () => {
        canvas.dataset.chartReady = 'true';
      });
      // Fallback: set ready after Chart.js animation via options.animation.onComplete
    </script>
    ```
